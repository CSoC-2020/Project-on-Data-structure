class Solution {
    public int findDuplicate(int[] nums) {
        Map<Integer,Integer> hm = new HashMap<>();
        for(int x : nums){
            if(!hm.containsKey(x))
                hm.put(x,1);
            else
                hm.put(x,hm.get(x) + 1);
        }
        for(int x : hm.keySet()){
            if(hm.get(x)>1)
                return x;
        }
        return 0;
    }
}
