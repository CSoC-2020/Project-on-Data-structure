class Solution {
    public boolean isHappy(int n) {
        if(n==1 || n==7)
            return true;
        int sum=n,x=0,temp=n;
        while(sum>9){
            sum=0;
            while(temp!=0){
                x=temp%10;
                sum=sum+(x*x);
                temp=temp/10;
            }
            temp=sum;
        }
        if(sum==1 || sum==7)
            return true;
        return false;
    }
}
