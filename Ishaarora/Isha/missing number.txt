class Solution {
public:
    int missingNumber(vector<int>& nums) { int i;
         if(nums.size()==0)
            return 0;
        sort(nums.begin(),nums.end());
         for(i=0;i<nums.size();i++)
        {
            if(nums[i]!=i)
            {
                return i;
            }
        }
      return nums.size();
    }
};
